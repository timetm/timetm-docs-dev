 51:             $firstDayOfMonth = date('Y-m-d', mktime(0, 0, 0, $calendar->getMonth(), 1, $calendar->getYear()));

 53:         }
 54:         elseif ($type == 'week') {

 57:         }

 60:             $lastDayOfMonth  = date('Y-m-d', mktime(0, 0, 0, $calendar->getMonth(), $calendar->getDay(), $calendar->getYear()));
 61:         }
 62: 
 63:         // get query builder
 64:         $queryBuilder = $this->em->createQueryBuilder();
 65: 
 66:         /*
 67:          * build and execute query
 68:          *
 69:          * select title,date
 70:          *   from Event e
 71:          *     join Agenda a on e.agenda_id =a.id
 72:          *     join fos_user u on a.user_id=u.id
 73:          *     where a.id=1; TODO
 74:         */
 75:         $events = $queryBuilder
 76:             ->select('partial e.{id, title, place, startdate , starttime }')
 77:             ->from('TimeTMCoreBundle:Event', 'e')
 78:             ->leftjoin('e.agenda', 'a')
 79:             ->leftjoin('a.user', 'u')
 80:             ->where('e.startdate BETWEEN :firstDay AND :lastDay')
 81:             ->setParameter('firstDay', $firstDayOfMonth)
 82:             ->setParameter('lastDay', $lastDayOfMonth)
 83:             ->getQuery()
 84:             ->execute();
 85: 
 86:         // add events to the dates array
 87:         foreach ( $dates as &$date ) {
 88:             if (isset($date['datestamp'])) {
 89:                 $date['events'] = array();
 90:                 foreach ( $events as $event ) {
 91:                     if ( $event->getStartdate()->format('Y/m/d') == $date['datestamp'] ) {
 92:                         array_push($date['events'], $event);
 93:                     }
 94:                 }
 95:             }
 96:         }
 97: 
 98:         return $dates;
